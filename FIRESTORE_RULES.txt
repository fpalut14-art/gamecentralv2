rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() { return request.auth != null; }
    function userPath() { return /databases/$(database)/documents/users/$(request.auth.uid); }
    function hasUser() { return signedIn() && exists(userPath()); }
    function userData() { return get(userPath()).data; }
    function isAdmin() { return hasUser() && userData().role == "admin" && userData().banned != true; }
    function isSeller() { return hasUser() && (userData().role == "seller" || userData().role == "admin") && userData().banned != true; }
    function isNotBanned() { return signedIn() && (!hasUser() || !("banned" in userData()) || userData().banned != true); }

    match /users/{userId} {
      allow read: if signedIn() && (request.auth.uid == userId || isAdmin());
      allow create: if signedIn() && request.auth.uid == userId;
      allow update: if signedIn() && (request.auth.uid == userId || isAdmin());
      allow delete: if isAdmin();
    }

    match /products/{productId} {
      allow read: if resource.data.status == "active" || isAdmin() || (signedIn() && request.auth.uid == resource.data.sellerId);
      allow create: if isSeller() && isNotBanned() && request.resource.data.status == "pending" && request.resource.data.sellerId == request.auth.uid;
      allow update, delete: if isAdmin();
    }

    match /orders/{orderId} {
      allow read: if isAdmin() || request.auth.uid == resource.data.buyerId || request.auth.uid == resource.data.sellerId;
      allow create: if signedIn() && isNotBanned() && request.resource.data.status == "pending_seller";
      allow update: if isAdmin() || request.auth.uid == resource.data.buyerId || request.auth.uid == resource.data.sellerId;
      allow delete: if isAdmin();
    }

    match /ads/{adId} {
      allow read: if true;
      allow create: if signedIn() && isNotBanned();
      allow update, delete: if isAdmin();
    }

    match /notifications/{id} {
      allow read, update: if signedIn() && (request.auth.uid == resource.data.userId || isAdmin());
      allow create: if signedIn();
      allow delete: if isAdmin();
    }

    match /chats/{id} {
      allow create: if signedIn() && isNotBanned() && request.auth.uid in request.resource.data.participants;
      allow read, update: if signedIn() && (isAdmin() || request.auth.uid in resource.data.participants);
      allow delete: if isAdmin();
    }

    match /messages/{id} {
      allow read: if signedIn();
      allow create: if signedIn() && isNotBanned() && request.resource.data.senderId == request.auth.uid;
      allow update, delete: if isAdmin();
    }

    match /support_tickets/{id} {
      allow create: if signedIn() && isNotBanned();
      allow read, update: if isAdmin() || request.auth.uid == resource.data.userId;
      allow delete: if isAdmin();
    }

    match /support_messages/{id} {
      allow read: if isAdmin() || signedIn();
      allow create: if isAdmin() || (signedIn() && isNotBanned());
      allow update, delete: if isAdmin();
    }

    match /reports/{id} {
      allow create: if signedIn() && isNotBanned();
      allow read, update, delete: if isAdmin();
    }

    match /logs/{id} {
      allow read, write: if isAdmin();
    }
  }
}
